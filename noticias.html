async function fetchNews() { try { const response = await fetch('/api/noticias'); const data = await response.json(); const newsContainer = document.getElementById('news-container'); newsContainer.innerHTML = ''; if (data.status === 'ok' && Array.isArray(data.articles) && data.articles.length > 0) { data.articles.forEach(article => { const newsItem = document.createElement('div'); newsItem.classList.add('news-item'); newsItem.innerHTML = `

${article.title}

${article.description || 'Descrição não disponível'}

`; newsContainer.appendChild(newsItem); }); } else { newsContainer.innerHTML = '

Não há notícias disponíveis no momento.

'; } } catch (error) { console.error('Erro ao buscar notícias:', error); const newsContainer = document.getElementById('news-container'); newsContainer.innerHTML = '

Erro ao carregar notícias. Tente novamente mais tarde.

'; } } fetchNews();
